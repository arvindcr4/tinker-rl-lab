          <pre class="python"><span class="hljs-string">"""
TRL Knowledge Distillation Implementation
==========================================
Port of Tinker Distillation experiments to TRL/transformers.

Two approaches:
1. Off-Policy: SFT on teacher outputs (OpenThoughts3 style)
2. On-Policy: KL divergence minimization to teacher
"""</span>

<span class="hljs-keyword">import</span> torch
<span class="hljs-keyword">import</span> torch.nn.functional <span class="hljs-keyword">as</span> F
<span class="hljs-keyword">from</span> typing <span class="hljs-keyword">import</span> List
<span class="hljs-keyword">from</span> dataclasses <span class="hljs-keyword">import</span> dataclass
<span class="hljs-keyword">from</span> datasets <span class="hljs-keyword">import</span> Dataset
<span class="hljs-keyword">from</span> transformers <span class="hljs-keyword">import</span> (
    AutoModelForCausalLM,
    AutoTokenizer,
    Trainer,
    TrainingArguments,
)
<span class="hljs-keyword">from</span> trl <span class="hljs-keyword">import</span> SFTTrainer, SFTConfig


<span class="hljs-meta">@dataclass</span>
<span class="hljs-class"><span class="hljs-keyword">class</span> <span class="hljs-title">DistillationConfig</span>:</span>
    <span class="hljs-string">"""Configuration for knowledge distillation."""</span>
    teacher_model_name: str = <span class="hljs-string">"meta-llama/Llama-3.2-3B-Instruct"</span>
    student_model_name: str = <span class="hljs-string">"meta-llama/Llama-3.2-1B"</span>
    temperature: float = <span class="hljs-number">2.0</span>
    alpha: float = <span class="hljs-number">0.5</span>
    max_length: int = <span class="hljs-number">512</span>


<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">generate_teacher_dataset</span><span class="hljs-params">(
    teacher_model,
    tokenizer,
    prompts: List[str],
    max_new_tokens: int = <span class="hljs-number">256</span>,
)</span> -&gt; Dataset:</span>
    <span class="hljs-string">"""Generate teacher outputs for off-policy distillation."""</span>
    teacher_model.eval()
    data = []

    <span class="hljs-keyword">for</span> prompt <span class="hljs-keyword">in</span> prompts:
        inputs = tokenizer(prompt, return_tensors=<span class="hljs-string">"pt"</span>).to(teacher_model.device)

        <span class="hljs-keyword">with</span> torch.no_grad():
            outputs = teacher_model.generate(
                **inputs,
                max_new_tokens=max_new_tokens,
                do_sample=<span class="hljs-literal">True</span>,
                temperature=<span class="hljs-number">0.7</span>,
                top_p=<span class="hljs-number">0.9</span>,
                pad_token_id=tokenizer.eos_token_id,
            )

        response = tokenizer.decode(
            outputs[<span class="hljs-number">0</span>][inputs.input_ids.shape[<span class="hljs-number">1</span>]:],
            skip_special_tokens=<span class="hljs-literal">True</span>
        )

        data.append({
            <span class="hljs-string">"prompt"</span>: prompt,
            <span class="hljs-string">"completion"</span>: response,
            <span class="hljs-string">"text"</span>: <span class="hljs-string">f"<span class="hljs-subst">{prompt}</span>\n<span class="hljs-subst">{response}</span>"</span>,
        })

    <span class="hljs-keyword">return</span> Dataset.from_list(data)


<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">train_off_policy_distillation</span><span class="hljs-params">(config: DistillationConfig, prompts: List[str])</span>:</span>
    <span class="hljs-string">"""Off-policy distillation: SFT on teacher outputs."""</span>
    print(<span class="hljs-string">"Loading teacher model..."</span>)
    teacher = AutoModelForCausalLM.from_pretrained(
        config.teacher_model_name,
        torch_dtype=torch.bfloat16,
        device_map=<span class="hljs-string">"auto"</span>,
    )
    tokenizer = AutoTokenizer.from_pretrained(config.teacher_model_name)
    tokenizer.pad_token = tokenizer.eos_token

    print(<span class="hljs-string">"Generating teacher dataset..."</span>)
    teacher_dataset = generate_teacher_dataset(teacher, tokenizer, prompts)

    <span class="hljs-keyword">del</span> teacher
    torch.cuda.empty_cache()

    print(<span class="hljs-string">"Loading student model..."</span>)
    student = AutoModelForCausalLM.from_pretrained(
        config.student_model_name,
        torch_dtype=torch.bfloat16,
        device_map=<span class="hljs-string">"auto"</span>,
    )

    sft_config = SFTConfig(
        output_dir=<span class="hljs-string">"./distillation_off_policy"</span>,
        per_device_train_batch_size=<span class="hljs-number">4</span>,
        gradient_accumulation_steps=<span class="hljs-number">4</span>,
        learning_rate=<span class="hljs-number">5e-5</span>,
        num_train_epochs=<span class="hljs-number">3</span>,
        max_seq_length=config.max_length,
        logging_steps=<span class="hljs-number">10</span>,
    )

    trainer = SFTTrainer(
        model=student,
        train_dataset=teacher_dataset,
        args=sft_config,
        tokenizer=tokenizer,
    )

    print(<span class="hljs-string">"Starting off-policy distillation..."</span>)
    trainer.train()
    trainer.save_model(<span class="hljs-string">"./distillation_off_policy_final"</span>)
    <span class="hljs-keyword">return</span> trainer


<span class="hljs-class"><span class="hljs-keyword">class</span> <span class="hljs-title">OnPolicyDistillationTrainer</span><span class="hljs-params">(Trainer)</span>:</span>
    <span class="hljs-string">"""On-policy distillation with KL divergence to teacher."""</span>

    <span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">__init__</span><span class="hljs-params">(self, teacher_model, temperature: float = <span class="hljs-number">2.0</span>, alpha: float = <span class="hljs-number">0.5</span>, **kwargs)</span>:</span>
        super().__init__(**kwargs)
        self.teacher_model = teacher_model
        self.temperature = temperature
        self.alpha = alpha

        <span class="hljs-keyword">for</span> param <span class="hljs-keyword">in</span> self.teacher_model.parameters():
            param.requires_grad = <span class="hljs-literal">False</span>
        self.teacher_model.eval()

    <span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">compute_loss</span><span class="hljs-params">(self, model, inputs, return_outputs=False, num_items_in_batch=None)</span>:</span>
        <span class="hljs-string">"""Combined loss: task loss + KL divergence."""</span>
        labels = inputs.pop(<span class="hljs-string">"labels"</span>, <span class="hljs-literal">None</span>)

        student_outputs = model(**inputs)
        student_logits = student_outputs.logits

        <span class="hljs-keyword">with</span> torch.no_grad():
            teacher_outputs = self.teacher_model(**inputs)
            teacher_logits = teacher_outputs.logits

        <span class="hljs-keyword">if</span> labels <span class="hljs-keyword">is</span> <span class="hljs-keyword">not</span> <span class="hljs-literal">None</span>:
            task_loss = F.cross_entropy(
                student_logits.view(<span class="hljs-number">-1</span>, student_logits.size(<span class="hljs-number">-1</span>)),
                labels.view(<span class="hljs-number">-1</span>),
                ignore_index=<span class="hljs-number">-100</span>,
            )
        <span class="hljs-keyword">else</span>:
            task_loss = torch.tensor(<span class="hljs-number">0.0</span>, device=student_logits.device)

        student_probs = F.log_softmax(student_logits / self.temperature, dim=<span class="hljs-number">-1</span>)
        teacher_probs = F.softmax(teacher_logits / self.temperature, dim=<span class="hljs-number">-1</span>)

        kl_loss = F.kl_div(student_probs, teacher_probs, reduction=<span class="hljs-string">"batchmean"</span>)
        kl_loss = kl_loss * (self.temperature ** <span class="hljs-number">2</span>)

        loss = self.alpha * task_loss + (<span class="hljs-number">1</span> - self.alpha) * kl_loss

        <span class="hljs-keyword">if</span> return_outputs:
            <span class="hljs-keyword">return</span> loss, student_outputs
        <span class="hljs-keyword">return</span> loss


<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">train_on_policy_distillation</span><span class="hljs-params">(config: DistillationConfig, train_dataset: Dataset)</span>:</span>
    <span class="hljs-string">"""On-policy distillation with KL minimization."""</span>
    print(<span class="hljs-string">"Loading teacher model..."</span>)
    teacher = AutoModelForCausalLM.from_pretrained(
        config.teacher_model_name,
        torch_dtype=torch.bfloat16,
        device_map=<span class="hljs-string">"auto"</span>,
    )

    print(<span class="hljs-string">"Loading student model..."</span>)
    student = AutoModelForCausalLM.from_pretrained(
        config.student_model_name,
        torch_dtype=torch.bfloat16,
        device_map=<span class="hljs-string">"auto"</span>,
    )

    tokenizer = AutoTokenizer.from_pretrained(config.student_model_name)
    tokenizer.pad_token = tokenizer.eos_token

    training_args = TrainingArguments(
        output_dir=<span class="hljs-string">"./distillation_on_policy"</span>,
        per_device_train_batch_size=<span class="hljs-number">4</span>,
        gradient_accumulation_steps=<span class="hljs-number">4</span>,
        learning_rate=<span class="hljs-number">5e-5</span>,
        num_train_epochs=<span class="hljs-number">3</span>,
        logging_steps=<span class="hljs-number">10</span>,
        fp16=<span class="hljs-literal">True</span>,
        remove_unused_columns=<span class="hljs-literal">False</span>,
    )

    trainer = OnPolicyDistillationTrainer(
        teacher_model=teacher,
        temperature=config.temperature,
        alpha=config.alpha,
        model=student,
        args=training_args,
        train_dataset=train_dataset,
        tokenizer=tokenizer,
    )

    print(<span class="hljs-string">"Starting on-policy distillation (KL minimization)..."</span>)
    trainer.train()
    trainer.save_model(<span class="hljs-string">"./distillation_on_policy_final"</span>)
    <span class="hljs-keyword">return</span> trainer


<span class="hljs-function"><span class="hljs-keyword">def</span> <span class="hljs-title">main</span><span class="hljs-params">()</span>:</span>
    config = DistillationConfig()

    prompts = [
        <span class="hljs-string">"Explain machine learning in simple terms."</span>,
        <span class="hljs-string">"What are the benefits of exercise?"</span>,
        <span class="hljs-string">"How does photosynthesis work?"</span>,
    ] * <span class="hljs-number">30</span>

    print(<span class="hljs-string">"="</span> * <span class="hljs-number">60</span>)
    print(<span class="hljs-string">"Tinker Distillation -&gt; TRL Implementation"</span>)
    print(<span class="hljs-string">"="</span> * <span class="hljs-number">60</span>)

    print(<span class="hljs-string">"\n--- Part 1: Off-Policy Distillation ---"</span>)
    train_off_policy_distillation(config, prompts)

    print(<span class="hljs-string">"\n--- Part 2: On-Policy Distillation ---"</span>)
    tokenizer = AutoTokenizer.from_pretrained(config.student_model_name)
    tokenizer.pad_token = tokenizer.eos_token

    dataset = Dataset.from_dict({<span class="hljs-string">"text"</span>: prompts})
    tokenized = dataset.map(
        <span class="hljs-keyword">lambda</span> x: tokenizer(x[<span class="hljs-string">"text"</span>], truncation=<span class="hljs-literal">True</span>, max_length=config.max_length, padding=<span class="hljs-string">"max_length"</span>),
        batched=<span class="hljs-literal">True</span>
    )
    train_on_policy_distillation(config, tokenized)

    print(<span class="hljs-string">"\nDistillation complete!"</span>)


<span class="hljs-keyword">if</span> __name__ == <span class="hljs-string">"__main__"</span>:
