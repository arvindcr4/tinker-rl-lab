  node distance=0.95cm and 1.0cm,
  block/.style={
    rectangle, rounded corners=4pt, draw=black!60, fill=gray!7,
    minimum width=2.5cm, minimum height=0.78cm, align=center,
    font=\small
  },
  reward/.style={
    rectangle, rounded corners=4pt, draw=orange!70!black, fill=orange!10,
    minimum width=2.6cm, minimum height=0.78cm, align=center,
    font=\small
  },
  update/.style={
    rectangle, rounded corners=4pt, draw=green!45!black, fill=green!9,

\node[block] (prompt) {Prompt batch};
\node[block, right=of prompt] (sample) {Sample $G$ responses\\per prompt};
\node[reward, right=of sample] (score) {Rule-based reward\\JSON, math, code};

\node[update, left=of loss] (policy) {Updated policy};

\draw[arrow] (prompt) -- (sample);
\draw[arrow] (sample) -- (score);

\draw[arrow] (adv) -- (loss);
\draw[arrow] (loss) -- (policy);
\draw[arrow, dashed] (policy.north) |- node[pos=0.25, above, font=\scriptsize] {next step} (sample.north);

\node[draw=red!60!black, fill=red!7, rounded corners=4pt, align=center,

\draw[-{Stealth[length=2.0mm]}, red!65!black, thick] (zvf.north) -- (adv.south);
